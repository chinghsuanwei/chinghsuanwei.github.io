Rakuten Codility online programming test

Only one question

Q1.
There has 3 rule for string

  1. a string must start with a consonant
  2. no two consecutive vowels
  3. no two consecutive consonants

At first time, I just wanted to use bruteforce-way to solve the problem. But I thought I could not pass the large test if I use bruteforce-way.
Then I spend like 20-min to figure out the question.

It’s just a permutations problem.

The count of vowels must equal or less than 1 to consonants.

long long factorial(int N)
{
long long sum = 1;
for(int i=1; i<=N; i++)
{
sum *= i;
}

return sum;

}

int solution(string &S) {
// write your code in C++14 (g++ 6.2.0)
std::sort(S.begin(), S.end());

std::unordered_map<char, int> vowels;
std::unordered_map<char, int> consonants;

int vowels_count = 0;
int consonants_count = 0;

for(int i=0; i<S.length(); i++)
{
    if(S[i] == 'A' || S[i] == 'E' || S[i] == 'I' || S[i] == 'O' || S[i] == 'U')
    {
        std::unordered_map<char,int>::iterator it = vowels.find(S[i]);

        // key already present in the map
        if (it != vowels.end()) {
            it->second++;    // increment map's value for key 'c'
        }
        // key not found
        else {
            vowels.insert(std::make_pair(S[i], 1));
        }
        vowels_count++;
    }
    else
    {
        std::unordered_map<char,int>::iterator it = consonants.find(S[i]);

        // key already present in the map
        if (it != consonants.end()) {
            it->second++;    // increment map's value for key 'c'
        }
        // key not found
        else {
            consonants.insert(std::make_pair(S[i], 1));
        }
        consonants_count++;
    }
}


if(vowels_count == consonants_count || (vowels_count+1) == consonants_count)
{
    long long vowel_sum = 1;
    for(int i=1; i<=vowels_count; i++)
    {
        vowel_sum *= i;
    }

    for (std::unordered_map<char,int>::iterator it=vowels.begin(); it!=vowels.end(); ++it)
    {
        vowel_sum /= factorial(it->second);
    }

    long long consonants_sum = 1;
    for(int i=1; i<=consonants_count; i++)
    {
        consonants_sum *= i;
    }

    for (std::unordered_map<char,int>::iterator it=consonants.begin(); it!=consonants.end(); ++it)
    {
        consonants_sum /= factorial(it->second);
    }


    return vowel_sum * consonants_sum;
}
else
{
    return 0;
}

}
